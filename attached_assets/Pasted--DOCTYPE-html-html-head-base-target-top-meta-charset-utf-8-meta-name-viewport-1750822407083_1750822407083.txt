  <script>
    // Formatação do telefone no padrão brasileiro
    $(document).ready(function() {
      $('#telefone').on('input', function() {
        let value = $(this).val().replace(/\D/g, '');
        if (value.length > 11) {
          value = value.slice(0, 11);
        }
        
        let formattedValue = '';
        if (value.length > 0) {
          formattedValue = '(' + value.slice(0, 2);
          if (value.length > 2) {
            formattedValue += ') ' + value.slice(2, 7);
            if (value.length > 7) {
              formattedValue += '-' + value.slice(7, 11);
            }
          }
        }
        
        $(this).val(formattedValue);
      });
    });
    
    // ID da planilha e da pasta do Drive
    const SHEET_ID = '1iKmNZ7hddwFCZWpPJ52diA-k1CUMeJ_qqMPUm0ztzG0';
    const FOLDER_ID = '1K5xlEGI8ti_i0_qrAElnSVcl4CMaf_-w';
    
    // URL do Google Apps Script que será criado
    const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbz0JrXfnvB2nYzyFUfcm72rMRNe7R92_4rWVZ0p5JzRSPEF2otw4dyXByM8aNPcAeTu/exec';
    
    $(document).ready(function() {
      $('#curriculo-form').on('submit', function(e) {
        e.preventDefault();
        
        // Mostrar mensagem de carregamento
        $('#loading').show();
        $('#curriculo-form').hide();
        
        // Obter dados do formulário
        const nome = $('#nome').val();
        // Campo de data de nascimento removido
        const email = $('#email').val();
        const telefone = $('#telefone').val();
        // Campo de área removido
        const curriculoFile = $('#curriculo')[0].files[0];
        
        // Verificar se o arquivo foi selecionado
        if (!curriculoFile) {
          $('#loading').hide();
          $('#error-message').text('Por favor, selecione um arquivo de currículo.').show();
          $('#curriculo-form').show();
          return;
        }
        
        // Ler o arquivo como base64 para envio
        const reader = new FileReader();
        reader.onload = function(e) {
          const fileData = e.target.result; // Dados do arquivo em base64
          
          // Criar FormData para enviar os dados
          const formData = new FormData();
          formData.append('nome', nome);
          // Campo de data de nascimento removido
          formData.append('email', email);
          formData.append('telefone', telefone);
          // Campo de área removido
          formData.append('curriculo', fileData);
          formData.append('sheetId', SHEET_ID);
          formData.append('folderId', FOLDER_ID);
        
          // Enviar dados para o Google Apps Script
          $.ajax({
            url: SCRIPT_URL,
            type: 'POST',
            data: formData,
            processData: false,
            contentType: false,
            success: function(response) {
              $('#loading').hide();
              
              try {
                // Tentar interpretar a resposta como JSON
                if (typeof response === 'string') {
                  response = JSON.parse(response);
                }
                
                if (response.success) {
                  $('#success-message').show();
                } else {
                  $('#error-message').text('Erro no processamento: ' + (response.error || 'Erro desconhecido')).show();
                  console.error('Erro retornado pelo servidor:', response.error);
                  $('#curriculo-form').show();
                }
              } catch (e) {
                $('#error-message').text('Erro ao interpretar resposta do servidor.').show();
                console.error('Erro ao processar resposta:', e, response);
                $('#curriculo-form').show();
              }
            },
            error: function(error) {
              $('#loading').hide();
              $('#error-message').text('Erro ao enviar formulário: ' + error.statusText).show();
              console.error('Erro na requisição AJAX:', error);
              $('#curriculo-form').show();
            }
          });
        };
        
        // Iniciar leitura do arquivo
        reader.readAsDataURL(curriculoFile);
      });
    });
  </script>
